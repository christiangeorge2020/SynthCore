371  inline bool smoothParameter(T in, T& out)
372  {
373  if (smootherType == smoothingMethod::kLPFSmoother)
374  {
375  z = (in * b) + (z * a);
376  if (z == z2)
377  {
378  out = in;

381  z2 = z;
382  out = z2;
383  return true;
384  }

386  {

390  return false;
391  }
392  if (in > z)
