72  bool processSynthAudio(SynthProcessorData* audioData)
73  {
74  // --- make sure we have input and outputs
75  if (audioData->numInputChannels == 0 || audioData->numOutputChannels == 0)

77 
78  // --- if MONO, no panning is applied to left channel
79  if (audioData->numInputChannels == 1 && audioData->numOutputChannels == 1)
80  audioData->outputs[0] = audioData->inputs[0] * gainRaw;
81  else if (audioData->numOutputChannels > 1)
82  // --- stereo, add left pan value
83  audioData->outputs[0] = audioData->inputs[0] * gainRaw * panLeftGain;
84 
85  // --- now process right channel
86  // --- monot -> stereo: copy left channel to right and apply panning gain
87  if (audioData->numInputChannels == 1 && audioData->numOutputChannels == 2)
88  audioData->outputs[1] = audioData->inputs[0] * gainRaw * panRightGain;
89 
90  // --- stereo to stereo
91  else if (audioData->numInputChannels == 2 && audioData->numOutputChannels == 2)
92  audioData->outputs[1] = audioData->inputs[1] * gainRaw * panRightGain;
93 
94  return true;
95  }
